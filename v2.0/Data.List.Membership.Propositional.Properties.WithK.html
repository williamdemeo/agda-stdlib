------------------------------------------------------------------------
-- Irrelevance

unique⇒irrelevant : ∀ {a} {A : Set a} {xs : List A} →
                    Unique xs → Irrelevant (_∈ xs)
unique⇒irrelevant = Membershipₛ.unique⇒irrelevant (P.setoid _) ≡-irrelevant
